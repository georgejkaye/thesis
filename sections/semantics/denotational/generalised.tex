\section{Denotational semantics for generalised circuits}

\begin{notation}
    Given a set \(A\) and a word \(\listvar{v} \in \natplus^\star\) of
    length \(n\), we write \(
    A^{\listvar{v}}
    \coloneqq
    A^{\listvar{v}(0)}
    \times
    A^{\listvar{v}(1)}
    \times
    \dots
    \times
    A^{\listvar{v}(n-1)}
    \).
\end{notation}
